# ── 環境變數 ──────────────────────────────────────────────────

# ── HTTP 客戶端 ───────────────────────────────────────────────
# 非同步 HTTP 客戶端，用於：
#   - Brave / Tavily / Exa / SerpApi 多來源搜尋

litellm==1.91.1

# ── HTML → 正文提取 ──────────────────────────────────────────
# 高品質 HTML 正文提取器（主力引擎）
# 支援 Markdown 輸出、表格提取、精準模式（favor_precision）
# 作為 HTTP 爬取後的首選文本提取器；提供 baseline fallback
trafilatura==2.1.0

# ── PDF 自動解析 ──────────────────────────────────────────────
# 專案內 `crawl4ai_pdf.py` 提供 PDF 偵測與抽取邏輯
# PyMuPDF 為主引擎，pypdf 為文字層 fallback
PyMuPDF==1.28.0
pypdf==6.14.2

# ── HTML 解析（fallback 用）────────────────────────────────────
# 當 trafilatura 提取失敗時的後備 HTML 解析器
# 也用於提取頁面標題（<title> / <h1>）
# lxml 作為 BeautifulSoup 的高速解析後端

# ── JS 渲染爬取（必要）────────────────────────────────────────
